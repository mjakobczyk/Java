package pl.polsl.java.michal.jakobczyk.lab2.controller;

import pl.polsl.java.michal.jakobczyk.lab2.model.*;
import pl.polsl.java.michal.jakobczyk.lab2.view.*;
import java.io.*;
import pl.polsl.java.michal.jakobczyk.lab2.exception.IncorrectConversionTypeException;

/**
 * Class for conversions betweens languages
 * @author Michał Jakóbczyk
 * @version 1.0
 */
public class ConverterController {
    
    /**
     * Converter model
     */
    private final ConverterModel converterModel;
    
    /**
     * Converter view
     */
    private final ConverterView converterView;
    
    /**
     * Constructor of the converter
     * @param fileInputName for textController
     * @param fileOutputName for textController
     * @param conversionType to know the conversion type
     */
    public ConverterController(String fileInputName, String fileOutputName, String conversionType) {
        this.converterModel = new ConverterModel(fileInputName, fileOutputName, conversionType);
        this.converterView = new ConverterView();
    }
    
    
    /**
     * Run method of the conversion
     * @throws IOException when there was a problem with i/o
     */
    public void run() throws IOException, IncorrectConversionTypeException {
        this.converterView.startConversion();
        this.converterModel.convert();
        this.converterView.endConversion();
    }
    
}
