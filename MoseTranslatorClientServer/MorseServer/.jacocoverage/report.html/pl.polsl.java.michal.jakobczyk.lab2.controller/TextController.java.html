package pl.polsl.java.michal.jakobczyk.lab2.controller;

import pl.polsl.java.michal.jakobczyk.lab2.model.*;
import java.io.*;

/**
 * Controller class for text
 * @author Michał Jakóbczyk
 * @version 1.0
 */ 
public class TextController {
    
    /**
     * Stores the text model
     */
    private final TextModel textModel;
    
    /**
     * Constructor of the text controller
     * @param fileInputName for the input directory
     * @param fileOutputName for the output directory
     */
    public TextController (String fileInputName, String fileOutputName) {
        this.textModel = new TextModel(fileInputName, fileOutputName);
    }
    
    /**
     * Getter for the text model
     * @return text model
     */
    public TextModel getTextModel() {
        return this.textModel;
    }
    
    /**
     * Enabled running of encapsulated method
     * @throws IOException when there was a problem with i/o
     */
    public void readData() throws IOException {
        this.textModel.readDataFromFile();
    }
    
    
    /**
     * Enabled running of encapsulated method
     * @throws IOException when there was a problem with i/o
     */
    public void writeData() throws IOException {
        this.textModel.writeDataToFile();
    }
}
