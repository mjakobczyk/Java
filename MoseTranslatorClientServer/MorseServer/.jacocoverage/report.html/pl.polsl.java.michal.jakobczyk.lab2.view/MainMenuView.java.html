package pl.polsl.java.michal.jakobczyk.lab2.view;

/**
 * Main menu view, shows useful information about the application
 * @author Michał Jakóbczyk
 * @version 1.0
 */
public class MainMenuView {
    
    /**
     * Guides the user how to use the program
     */
    public void showHelp() {
        System.out.println("****** MORSE TRANSLATOR ******");
        System.out.println("> Converter Morse -> Latin / Latin -> Morse");
        System.out.println("> Program takes 3 parameters in the given order:");
        System.out.println("> <input_file_name> " + "<output_file_name> " + "<conversion_type>");
        System.out.println("> Example file name: C:\\\\Dokumenty\\\\Pliki\\\\plikwej.txt");
        System.out.println("> Possible conversion types: m2l (Morse to Latin) or l2m (Latin to Morse)\n");
    }
    
    /**
     * Asks the user for the name of the input file
     * containing text file to convert
     */
    public void askForInputFileName() {
        System.out.println("Enter full input file name with directory: ");
    }
    
    /**
     * Warns the user about an incorrect input file name
     */
    public void incorrectInputFileName() {
        System.out.println("Incorrect input file name! "
                + "Remember about double backslash '\\\\' instead of single backslash '\\'!\n"
                + "Hint: Example file name: C:\\\\Dokumenty\\\\Pliki\\\\plikwej.txt\n");
    }
    
    /**
     * Asks the user for the name of the output file
     * to save the converted text from the input file
     */
    public void askForOutputFileName() {
        System.out.println("Enter full output file name with directory: ");
    }
    
    /**
     * Warns the user about an incorrect output file name
     */
    public void incorrectOutputFileName() {
        System.out.println("Incorrect output file name! "
                + "Remember about double backslash '\\\\' instead of single backslash '\\'!\n"
                + "Hint: Example file name: C:\\\\Dokumenty\\\\Pliki\\\\plikwej.txt\n");
    }
    
    /**
     * Asks the user for the conversion type
     * from the available types (morse to latin
     * or latin to morse)
     */
    public void askForConversionType() {
        System.out.println("Choose conversion type (either 'm2l' or 'l2m'): ");
    }
    
    /*
    Warns the user about an incorrect conversion type
    */
    public void incorrectConversionType() {
        System.out.println("Incorrect conversion type! "
               + "Hint: Possible conversion types: m2l (Morse to Latin) or l2m (Latin to Morse)\n");
    }
    
     /**
     * Informs the user that the application is closing down
     */
    public void turnOffMessage() {
        System.out.println("Application is turning off.\n");
    }

}
