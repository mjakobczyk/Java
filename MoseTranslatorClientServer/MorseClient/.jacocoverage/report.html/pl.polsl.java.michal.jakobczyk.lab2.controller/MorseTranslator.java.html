package pl.polsl.java.michal.jakobczyk.lab2.controller;

import pl.polsl.java.michal.jakobczyk.lab2.exception.*;
import java.io.*;

/**
 * The entry point of the apllication. Contains main method.
 * @author Michał Jakóbczyk
 * @version 1.0
 */
public class MorseTranslator {
    
    /**
    * Starting method. 
    * @throws IncorrectConversionTypeException if the type was unknown
    * @throws IOException if there was a problem with i/o
    * @throws ArrayIndexOutOfBoundsException when no arguments passed
    * @param args[0] - name of input file
    * @param args[1] - name of output file
    * @param args[2] - conversion type
    */
    
    public static void main(String args[]) 
            throws IncorrectConversionTypeException, IOException, ArrayIndexOutOfBoundsException {
    
        try {
            MainMenuController mainMenuController;
            // If there were not all arguments passed
            if (args.length < 3)
                mainMenuController = new MainMenuController();
            // If exactly 3 arguments were passed
            else if (args.length == 3)
                mainMenuController = new MainMenuController(args[0], args[1], args[2]);
            // If other than that arguments were passed
            else
                throw new ArrayIndexOutOfBoundsException();
            
            mainMenuController.run();
        }
        catch (IncorrectConversionTypeException e) {
            System.out.println(e.getMessage());
        } 
        catch (IOException | ArrayIndexOutOfBoundsException e ) {
            System.out.println(e.getMessage());
        }
    }
    
}
